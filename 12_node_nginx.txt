  listen       80;
  server_name  ubuntuserver;

  #charset koi8-r;

  location / {
    #proxy_set_header Host $host;

    break;

# this file will be automatically loaded by nginx as long as the default

sudo service nginx reload


# stop/start nginx
sudo service nginx stop
sudo service nginx start
