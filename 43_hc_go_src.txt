# Get the Go source code (optional)
cd
mkdir src
cd src
git clone https://github.com/hectorcorrea/hectorcorrea.com.git

# Compile it
cd hectorcorrea
go get github.com/go-sql-driver/mysql
go build

# Create the MySQL database
mysql -u root < misc/createdb.sql

# Create a local user
CREATE USER 'bloguser'@'localhost' IDENTIFIED BY 'the-password';

GRANT SELECT,INSERT,UPDATE,DELETE,CREATE,DROP ON blogdb.* TO 'bloguser'@'localhost';
